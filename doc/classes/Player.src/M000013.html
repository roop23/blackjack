  def check_blackjack()
    hand = @hands[0]
    if hand.is_blackjack()
      print_player_hands()
      puts "Player #{@position} has a blackjack. You win 3:2!"
      add_winnings(hand, blackjack=true)                # Add the winnings of the bet to the player
      @current_hand += 1
    end
  end
