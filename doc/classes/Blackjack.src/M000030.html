  def init_table()
    puts "Please enter the number of players"
    valid_value = false
    while !valid_value
      value = gets.to_i
      if value > 0
        valid_value = true
      else
        puts "Please enter a valid number"
      end
    end
    @num_players = value
    @num_players.times do |index|
      add_player(Initial_Money, index)
    end
  end
